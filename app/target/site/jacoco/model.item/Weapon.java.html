 package model.item;

import model.character.Stats;

/**
 *
 * @author Toasty Studios
 */
public class Weapon extends Item {

    private Stats stats;
    private int damage;

    /**
     * init
     * @param name
     * @param description
     * @param value
     * @param rarity
     * @param unique
     * @param level
     * @param damage 
     */
    public Weapon(String name, String description, int value, String rarity, boolean unique, int level, int damage) {
        super(name, description, value, rarity, unique, level);
        this.damage = damage;
    }
    
    /**
     * Returns the stat bonus of the weapon
     * @return 
     */
    public Stats stats() {
        return this.stats;
    }

}
